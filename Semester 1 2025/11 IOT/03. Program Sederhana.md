```cpp
#define ECHO_PIN 10

  Serial.begin(9600);
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  Serial.println("📏 Ultrasonic Sensor Ready!");
}

void loop() {
  long duration, distance;
  
  // Kirim pulse
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  // Baca echo
  duration = pulseIn(ECHO_PIN, HIGH);
  distance = duration * 0.034 / 2;
  
  Serial.print("📐 Jarak: ");
  Serial.print(distance);
  Serial.println(" cm");
  
  delay(500);
}
```

```cpp
#define READ_BUTTON 3

int counter = 0;

  pinMode(WRITE_BUTTON, INPUT_PULLUP);
  pinMode(READ_BUTTON, INPUT_PULLUP);
  Serial.begin(9600);
  Serial.println("💾 EEPROM Demo Ready!");
  Serial.println("Press Button 2 to WRITE, Button 3 to READ");
}

void loop() {
  // Write to EEPROM
  if (digitalRead(WRITE_BUTTON) == LOW) {
    counter++;
    EEPROM.write(address, counter);
    Serial.print("💾 WRITE: ");
    Serial.println(counter);
    delay(500);
  }
  
  // Read from EEPROM
  if (digitalRead(READ_BUTTON) == LOW) {
    int value = EEPROM.read(address);
    Serial.print("📖 READ: ");
    Serial.println(value);
    delay(500);
  }
}
```

```cpp
#define BUTTON_PIN 2

    digitalWrite(LED_PIN, LOW);
    
    Serial.print("⏱️ Reaction Time: ");
    Serial.print(reactionTime);
    Serial.println(" ms");
    
    waiting = false;
    delay(1000); // Tunggu sebelum round berikutnya
  }
```

### 16. 🔄 Rotary Encoder dengan LED Position
```cpp
// 🔰 Project 16: Rotary Encoder Control
// 📍 GPIO2: CLK, GPIO3: DT, GPIO4: SW, GPIO5-9: LEDs

#define CLK 2  // Pin untuk clock signal encoder
#define DT 3   // Pin untuk data signal encoder  
#define SW 4   // Pin untuk switch/tombol encoder

int currentStateCLK;
int lastStateCLK;
int counter = 0;        // Menyimpan nilai counter encoder
String currentDir = ""; // Menyimpan arah putaran
unsigned long lastButtonPress = 0; // Debouncing button

int ledPins[] = {5, 6, 7, 8, 9}; // Pin untuk LED indicator
int ledCount = 5;

void setup() {
  // Set pin mode untuk encoder
  pinMode(CLK, INPUT);
  pinMode(DT, INPUT);
  pinMode(SW, INPUT_PULLUP);
  
  // Set pin mode untuk semua LED
  for(int i = 0; i < ledCount; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
  
  Serial.begin(9600);
  lastStateCLK = digitalRead(CLK); // Baca state awal
  Serial.println("🔄 Rotary Encoder Ready!");
}

void loop() {
  // Baca rotary encoder
  currentStateCLK = digitalRead(CLK);
  
  // Deteksi perubahan state pada pin CLK
  if (currentStateCLK != lastStateCLK) {
    // Jika pin DT berbeda dengan CLK, berarti putaran searah jarum jam
    if (digitalRead(DT) != currentStateCLK) {
      counter++; // Tambah counter
      currentDir = "CW"; // Clockwise
    } else {
      counter--; // Kurangi counter  
      currentDir = "CCW"; // Counter Clockwise
    }
    
    // Batasi nilai counter antara 0-4
    if (counter < 0) counter = 0;
    if (counter > 4) counter = 4;
    
    Serial.print("Direction: ");
    Serial.print(currentDir);
    Serial.print(" | Counter: ");
    Serial.println(counter);
    
    // Update LED position
    updateLEDs();
  }
  
  lastStateCLK = currentStateCLK; // Simpan state terakhir
  
  // Baca tombol encoder
  int btnState = digitalRead(SW);
  if (btnState == LOW) {
    // Debouncing - pastikan bukan noise
    if (millis() - lastButtonPress > 50) {
      Serial.println("🔄 Button Pressed - Reset!");
      counter = 0; // Reset counter
      updateLEDs(); // Update LED
    }
    lastButtonPress = millis();
  }
  
  delay(1); // Delay kecil untuk stabilisasi
}

void updateLEDs() {
  // Matikan semua LED terlebih dahulu
  for(int i = 0; i < ledCount; i++) {
    digitalWrite(ledPins[i], LOW);
  }
  // Nyalakan LED sesuai posisi counter
  digitalWrite(ledPins[counter], HIGH);
}
```

```cpp
// 🔰 Project 17: Infrared Remote Control
// 📍 GPIO11: IR Receiver, GPIO5-7: RGB LED

#include <IRremote.h>

#define IR_RECEIVE_PIN 11  // Pin receiver IR
#define RED_PIN 5          // Pin LED merah
#define GREEN_PIN 6        // Pin LED hijau  
#define BLUE_PIN 7         // Pin LED biru

void setup() {
  Serial.begin(9600);
  IrReceiver.begin(IR_RECEIVE_PIN, ENABLE_LED_FEEDBACK); // Start IR receiver
  
  // Set pin mode untuk RGB LED
  pinMode(RED_PIN, OUTPUT);
  pinMode(GREEN_PIN, OUTPUT);
  pinMode(BLUE_PIN, OUTPUT);
  
  Serial.println("📡 IR Remote Control Ready!");
  Serial.println("1: RED, 2: GREEN, 3: BLUE, 4: WHITE, 5: OFF");
}

void loop() {
  if (IrReceiver.decode()) { // Cek apakah ada data IR diterima
    unsigned long command = IrReceiver.decodedIRData.command; // Ambil kode command
    
    Serial.print("📶 IR Command Received: ");
    Serial.println(command);
    
    // Proses command berdasarkan kode remote
    switch(command) {
      case 12: // Button 1 - Merah
        setRGBColor(255, 0, 0);
        Serial.println("🔴 RED LED");
        break;
      case 24: // Button 2 - Hijau
        setRGBColor(0, 255, 0);
        Serial.println("🟢 GREEN LED");
        break;
      case 94: // Button 3 - Biru
        setRGBColor(0, 0, 255);
        Serial.println("🔵 BLUE LED");
        break;
      case 8:  // Button 4 - Putih
        setRGBColor(255, 255, 255);
        Serial.println("⚪ WHITE LED");
        break;
      case 28: // Button 5 - Mati
        setRGBColor(0, 0, 0);
        Serial.println("⚫ LED OFF");
        break;
      default:
        Serial.println("❓ Unknown Command");
    }
    
    IrReceiver.resume(); // Siap menerima signal berikutnya
  }
  delay(100);
}

// Fungsi untuk mengatur warna RGB LED
void setRGBColor(int red, int green, int blue) {
  analogWrite(RED_PIN, red);
  analogWrite(GREEN_PIN, green);
```

### 18. 🎵 Music Player dengan Buzzer
```cpp
// 🔰 Project 18: Simple Music Player
// 📍 GPIO8: Buzzer, GPIO2-4: Control Buttons

#include "pitches.h" // Library untuk frekuensi nada

#define BUZZER_PIN 8
#define BTN_PLAY 2
#define BTN_STOP 3
#define BTN_NEXT 4

// Melodi "Twinkle Twinkle Little Star"
int melody[] = {
  NOTE_C4, NOTE_C4, NOTE_G4, NOTE_G4, NOTE_A4, NOTE_A4, NOTE_G4,
  NOTE_F4, NOTE_F4, NOTE_E4, NOTE_E4, NOTE_D4, NOTE_D4, NOTE_C4
};

// Durasi setiap nada (4 = quarter note, 8 = eighth note)
int noteDurations[] = {
  4, 4, 4, 4, 4, 4, 2,
  4, 4, 4, 4, 4, 4, 2
};

bool isPlaying = false;
int currentNote = 0;

void setup() {
  pinMode(BTN_PLAY, INPUT_PULLUP);
  pinMode(BTN_STOP, INPUT_PULLUP);
  pinMode(BTN_NEXT, INPUT_PULLUP);
  
  Serial.begin(9600);
  Serial.println("🎵 Music Player Ready!");
  Serial.println("Play: Button 2, Stop: Button 3, Next: Button 4");
}

void loop() {
  // Cek tombol play
  if (digitalRead(BTN_PLAY) == LOW) {
    isPlaying = true;
    Serial.println("▶️ Playing Music...");
    delay(300); // Debouncing
  }
  
  // Cek tombol stop  
  if (digitalRead(BTN_STOP) == LOW) {
    isPlaying = false;
    noTone(BUZZER_PIN); // Stop buzzer
    Serial.println("⏹️ Music Stopped");
    delay(300);
  }
  
  // Cek tombol next
  if (digitalRead(BTN_NEXT) == LOW) {
    currentNote = 0; // Reset ke awal lagu
    Serial.println("⏭️ Song Reset");
    delay(300);
  }
  
  // Mainkan musik jika sedang play
  if (isPlaying) {
    playMusic();
  }
}

void playMusic() {
  // Hitung durasi nada berdasarkan tempo
  int noteDuration = 1000 / noteDurations[currentNote];
  
  // Mainkan nada saat ini
  tone(BUZZER_PIN, melody[currentNote], noteDuration);
  
  Serial.print("🎵 Note ");
  Serial.print(currentNote + 1);
  Serial.print(": ");
  Serial.println(melody[currentNote]);
  
  // Delay antara nada (ditambah 30% untuk pause)
  int pauseBetweenNotes = noteDuration * 1.30;
  delay(pauseBetweenNotes);
  
  // Stop buzzer untuk nada berikutnya
  noTone(BUZZER_PIN);
  
  // Pindah ke nada berikutnya
  currentNote++;
  
  // Jika sudah akhir lagu, ulang dari awal
  if (currentNote >= sizeof(melody) / sizeof(melody[0])) {
    currentNote = 0;
    Serial.println("🔁 Restarting Song...");
  }
}
```

### 19. 🌈 Color Spectrum dengan RGB LED
```cpp
// 🔰 Project 19: Smooth Color Spectrum Cycle
// 📍 GPIO9: Red, GPIO10: Green, GPIO11: Blue

int redPin = 9;
int greenPin = 10;
int bluePin = 11;

float hue = 0;        // Nilai hue (0-360 derajat)
float saturation = 1; // Saturasi warna (0-1)
float brightness = 0.5; // Kecerahan (0-1)

void setup() {
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
  
  Serial.begin(9600);
  Serial.println("🌈 Color Spectrum Cycle Ready!");
  Serial.println("Changing colors through HSV spectrum...");
}

void loop() {
  // Konversi HSV ke RGB
  HSVtoRGB(hue, saturation, brightness);
  
  // Tingkatkan hue untuk perubahan warna
  hue += 0.5;
  if (hue >= 360) {
    hue = 0; // Kembali ke awal setelah 360 derajat
  }
  
  // Tampilkan nilai hue di Serial Monitor
  Serial.print("🎨 Hue: ");
  Serial.println(hue);
  
  delay(50); // Kecepatan perubahan warna
}

// Fungsi konversi HSV to RGB
void HSVtoRGB(float h, float s, float v) {
  float r, g, b;
  
  int i = int(h / 60) % 6;
  float f = (h / 60) - i;
  float p = v * (1 - s);
  float q = v * (1 - f * s);
  float t = v * (1 - (1 - f) * s);
  
  switch(i) {
    case 0: r = v; g = t; b = p; break;
    case 1: r = q; g = v; b = p; break;
    case 2: r = p; g = v; b = t; break;
    case 3: r = p; g = q; b = v; break;
    case 4: r = t; g = p; b = v; break;
    case 5: r = v; g = p; b = q; break;
  }
  
  // Konversi ke nilai 0-255 dan tulis ke LED
  analogWrite(redPin, r * 255);
  analogWrite(greenPin, g * 255);
  analogWrite(bluePin, b * 255);
}
```

### 20. 🎯 Reaction Game dengan Multiple LEDs
```cpp
// 🔰 Project 20: Advanced Reaction Game
// 📍 GPIO2: Start Button, GPIO3: Reaction Button, GPIO5-9: Level LEDs

#define START_BUTTON 2
#define REACTION_BUTTON 3
#define LED_LEVEL_1 5
#define LED_LEVEL_2 6
#define LED_LEVEL_3 7
#define LED_LEVEL_4 8
#define LED_LEVEL_5 9

int levelLEDs[] = {LED_LEVEL_1, LED_LEVEL_2, LED_LEVEL_3, LED_LEVEL_4, LED_LEVEL_5};
int currentLevel = 1;
unsigned long reactionTimes[5]; // Simpan waktu reaksi setiap level
bool gameActive = false;
unsigned long startTime;

void setup() {
  pinMode(START_BUTTON, INPUT_PULLUP);
  pinMode(REACTION_BUTTON, INPUT_PULLUP);
  
  // Set semua LED sebagai output
  for(int i = 0; i < 5; i++) {
    pinMode(levelLEDs[i], OUTPUT);
  }
  
  Serial.begin(9600);
  Serial.println("🎯 Advanced Reaction Game!");
  Serial.println("Press START button to begin!");
  showStartAnimation();
}

void loop() {
  // Cek tombol start
  if (!gameActive && digitalRead(START_BUTTON) == LOW) {
    startGame();
    delay(500); // Debouncing
  }
  
  // Jika game aktif, cek tombol reaksi
  if (gameActive && digitalRead(REACTION_BUTTON) == LOW) {
    recordReactionTime();
    delay(300); // Debouncing
  }
}

void startGame() {
  gameActive = true;
  currentLevel = 1;
  Serial.println("🚀 Game Started! Get ready...");
  
  // Mainkan setiap level
  for(int level = 1; level <= 5; level++) {
    if (!gameActive) break; // Jika game dihentikan
    
    playLevel(level);
  }
  
  // Tampilkan hasil akhir
  if (gameActive) {
    showFinalResults();
  }
  
  gameActive = false;
  Serial.println("\n🎮 Game Over! Press START to play again.");
}

void playLevel(int level) {
  currentLevel = level;
  
  // Tampilkan level saat ini
  updateLevelDisplay();
  Serial.print("\n🎯 Level ");
  Serial.println(level);
  Serial.println("Wait for the LED...");
  
  // Delay acak antara 1-3 detik
  randomSeed(analogRead(0));
  long randomDelay = random(1000, 3000);
  delay(randomDelay);
  
  // Nyalakan LED level dan catat waktu
  digitalWrite(levelLEDs[level-1], HIGH);
  startTime = millis();
  Serial.println("💡 LED ON - PRESS NOW!");
  
  // Tunggu sampai tombol ditekan (timeout 3 detik)
  unsigned long waitStart = millis();
  while(digitalRead(REACTION_BUTTON) == HIGH) {
    if (millis() - waitStart > 3000) {
      // Timeout
      Serial.println("⏰ TIME OUT! Too slow...");
      digitalWrite(levelLEDs[level-1], LOW);
      gameActive = false;
      return;
    }
  }
}

void recordReactionTime() {
  if (!gameActive) return;
  
  unsigned long reactionTime = millis() - startTime;
  reactionTimes[currentLevel-1] = reactionTime;
  
  // Matikan LED level
  digitalWrite(levelLEDs[currentLevel-1], LOW);
  
  Serial.print("⏱️ Reaction Time: ");
  Serial.print(reactionTime);
  Serial.println(" ms");
  
  // Beri feedback berdasarkan kecepatan
  if (reactionTime < 200) {
    Serial.println("🎉 EXCELLENT! Lightning fast!");
  } else if (reactionTime < 400) {
    Serial.println("👍 Great job!");
  } else if (reactionTime < 600) {
    Serial.println("😊 Good!");
  } else {
    Serial.println("🐢 You can do better!");
  }
  
  delay(1000); // Beri jeda sebelum level berikutnya
}

void updateLevelDisplay() {
  // Matikan semua LED terlebih dahulu
  for(int i = 0; i < 5; i++) {
    digitalWrite(levelLEDs[i], LOW);
  }
  // Nyalakan LED untuk level yang sudah diselesaikan
  for(int i = 0; i < currentLevel-1; i++) {
    digitalWrite(levelLEDs[i], HIGH);
  }
}

void showFinalResults() {
  Serial.println("\n🏆 FINAL RESULTS:");
  Serial.println("==================");
  
  unsigned long totalTime = 0;
  for(int i = 0; i < 5; i++) {
    Serial.print("Level ");
    Serial.print(i+1);
    Serial.print(": ");
    Serial.print(reactionTimes[i]);
    Serial.println(" ms");
    totalTime += reactionTimes[i];
  }
  
  Serial.print("🎯 Average: ");
  Serial.print(totalTime / 5);
  Serial.println(" ms");
  
  // Victory animation
  victoryAnimation();
}

void victoryAnimation() {
  for(int i = 0; i < 3; i++) {
    for(int j = 0; j < 5; j++) {
      digitalWrite(levelLEDs[j], HIGH);
    }
    delay(200);
    for(int j = 0; j < 5; j++) {
      digitalWrite(levelLEDs[j], LOW);
    }
    delay(200);
  }
}

void showStartAnimation() {
  // Animation saat startup
  for(int i = 0; i < 5; i++) {
    digitalWrite(levelLEDs[i], HIGH);
    delay(150);
  }
  for(int i = 0; i < 5; i++) {
    digitalWrite(levelLEDs[i], LOW);
    delay(150);
  }
}
```

- 7x Resistors (5x 220Ω untuk LED, 2x 10kΩ untuk button)
